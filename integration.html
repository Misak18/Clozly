Примеры кода для быстрой интеграции онлайн примерочной в ваш магазин.

// 1. Подключение виджета Clozly на ваш сайт
(function() {
  // URL из главной страницы(ключ в ссылке и API разные!)
  const BASE_URL = 'https://clozly.ru/telegram-tryon.html?tryon_id=ВАШ-КЛЮЧ';

  // Кнопка инициализации примерки
  const btn = document.createElement('button');
  btn.className = 'clozly-btn';
  btn.innerText = 'Примерить онлайн';
  
  btn.onclick = () => {
    // Считываем артикул товара из URL
    //В виде числа передается артикул товара для автоматического выбора товара.
    //В личном кабинете важно добавлять этот артикул c помощью редактора товара/при созданий товара
    const path = window.location.pathname.split('/');
    const sku = path[path.length - 1] || 'default';

    // Открываем фрейм примерки
    const iframe = document.getElementById('clozly-iframe');
    iframe.src = `${BASE_URL}&sku=${sku}`;
    document.getElementById('clozly-modal').style.display = 'flex';
  };

  document.querySelector('.product-actions').appendChild(btn);
})();
import Clozly from 'clozly';

  userPhotoUrl: "https://clozly.ru/img/False_img.png", // исходное фото человека
  useBackView: false // вид спереди (false) или сзади (true)
});
